question 1

11、请用 js 程序实现二分算法。

答：

function binaryFind(arr, num, start, end) {
    start = start || 0;
    end = end || arr.length;
    var mid = Math.floor((start + end) / 2);
    var midVal = arr[mid];
    if (start >= end) {
        return false;
    }
    if (midVal === num) {
        return 'arr[' + mid + '] = ' + midVal;
    } else {
        if (midVal > num) {
            return binaryFind(arr, num, 0, mid);
        } else {
            return binaryFind(arr, num, mid+1);
        }
    }
}

question 2

12、有这样一个 URL：http://vip.qq.com/a.php?a=1&b...，请写一段 JS 程序提取 URL 中的各个参数（参数名和参数个数不确定），将其按 key-value 形式返回到一个 json 结构中，如{a:’1’,b:’2’,c:’’,d:’xxx’,e:undefined}。

答：

/**
 * 解析url参数为对象 
 * @param  {[type]} url [description]
 * @return {[type]}     [description]
 */
function parseUrl(url) {
    url = url.toString();
    var ret = {};
    var urlArr = url.split('?');
    if (urlArr.length < 2) {
        return ret;
    }
    var paramsArr = urlArr[1].split('&');
    var len = paramsArr.length;
    for (var i=0; i<len; i++) {
        var arr = paramsArr[i].split('=');
        ret[arr[0]] = arr.length > 1 ? arr[1] : undefined;
    }
    return ret;
}

question 4

5、统计字符串中字母个数或统计最多字母数。
答：

var str = "liann";
var obj = {};
for(let i = 0;i<str.length;i++){
  var char = str.charAt(i);
  if(obj[char]===undefined){
    obj[char]={};
    obj[char].count=1;
  }else{
    obj[char].count++;
  }
}
Object.keys(obj).map(function(item){
  console.log(item,obj[item].count);
})
