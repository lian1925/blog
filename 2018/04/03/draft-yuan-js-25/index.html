列表类

class List{
  constructor(array){
    if(array instanceof Array){
      this._length = array.length;
      this._data = [];
      array.map((item)=>this._data.push(item))
    }else{
      this._data = [];
      this._length = 0;
    }
  }

  add(item){
    this._data.push(item);
    this._length ++;
  }
}

统计 字符串中重复最多的字母

var a = "lilannn";

var obj = {};

for(let i=0; i<a.length;i++){
  let char = a.charAt(i);
  if(obj[char] === undefined){
    obj[char]={};
    obj[char].count=1;
  }else{
    obj[char].count++;
  }
}

var result = Object.keys(obj).sort((a,b)=>{
  return obj[a].count-obj[b].count;
});

解析url

var parseUrl = function(url){
  url = url.toString();
  var result = {};

  var urlArr = url.split('?');
  if(urlArr.length<2){
    return result;
  }
  var paramsArr = urlArr[1].split('&');
  for(let i=0;i<paramsArr.length;i++){
    var arr = paramsArr[i].split('=');
    result[arr[0]] = arr.length>1?arr[1]:undefined
  }
  return result;
}
