课6 Express与MySQL

作者： Lian

日期： 2017年11月08日 17:11

本节将学习Express，并对MySQL进行数据的增删改查操作。

新知识

本课我们将学习两个新的Node.js库，并初步接触 Restful的API写法。

co 库，对JavaScript程序进行异步操作。

mysql2 库，用于数据库的操作，包括常用的对数据表增删改查。

代码练习

var express = require('express');
var router = express.Router();
var co = require('co');
var mysql = require('mysql2');
var app = express();

var mysql_info = {
  // connectionLimit : 500,
  host     : '127.0.0.1',
  user     : 'root',
  password : '',
  database : 'test'
};

app.get('/', function(req, res){
    co(function * () {
      var conn = yield mysql.createConnectionPromise(mysql_info);
    var query = "SELECT * FROM user" ;
      var data = yield conn.query(query);
      res.send({
          result:0,
          data:data[0]
      })
    })
});
app.put('/', function(req, res){
    var params = req.body
    co(function * () {
    var conn = yield mysql.createConnectionPromise(mysql_info);
      var query = "INSERT INTO goods SET ?";
      var data = yield conn.query(query,params)
      res.send({
         result:0,
          data:data[0]
      })
    })
});
app.post('/:id', function(req, res){
    var id = req.params.id;
    co(function * () {
    var conn = yield mysql.createConnectionPromise(mysql_info);
    var post = params;
    var query = "UPDATE goods SET ?"+" WHERE id=" + id ;
    var data = yield conn.query(query,post)
      res.send({
          result:0,
          data:data[0]
      })
    })
});
app.delete('/:id', function(req, res){
    var id = req.params.id;
    co(function * () {
    var conn = yield mysql.createConnectionPromise(mysql_info);
    var query = "DELETE FROM goods"+" WHERE id=" + id ;
    var data = yield conn.query(query)
      res.send({
          result:0
      })
    })
});

var server = app.listen(3000, function () {
  var host = server.address().address;
  var port = server.address().port;

  console.log('Example app listening at http://%s:%s', host, port);
});

常见问题

1、为什么请求返回json数据？
因为json数据，简单高效，容易解析，是当今程序员最喜欢的数据交换格式。

2、为什么使用co库？
co库是nodejs领域鼎鼎大名的TJ大神的代表作，用在数据库操作上，可以实现将异步转同步，提高代码可读性。
